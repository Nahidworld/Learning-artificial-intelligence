f=open("graph.txt")
vertex_and_edges=f.readline()
vertex = int(vertex_and_edges.split()[0])
edges= int(vertex_and_edges.split()[1])

#create empty matrix
graph=[]
for i in range(vertex):
    row=[0]*vertex
    graph.append(row)
    
#populate graph data in matrix
for line in f.readlines():
    u= int(line.split()[0])
    v= int(line.split()[1])
    graph[u][v]=1
    graph[v][u]=1

for row in graph:
    print(row)
